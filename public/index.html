async function verGrafico(){ let ticker = document.getElementById('ticker').value; document.getElementById('saida').innerText = 'Buscando ' + ticker + '...'; document.getElementById('grafico').src = ''; let r = await fetch('/api/grafico?ticker=' + ticker); let data = await r.json(); if(data.url){ // MUDEI DE 'imagem' PRA 'url' document.getElementById('saida').innerText = data.texto; document.getElementById('grafico').src = data.url; // URL DIRETO, SEM BASE64 } else { document.getElementById('saida').innerText = data.texto || 'Erro ao gerar gráfico'; } }
